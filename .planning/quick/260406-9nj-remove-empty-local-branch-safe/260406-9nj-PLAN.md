    ```typescript
    export async function gitCommitsAheadOf(branch: string, base: string): Promise<string[]> {
      try {
        const { stdout } = await execa('git', ['log', `${base}..${branch}`, '--oneline']);
        return stdout.trim() ? stdout.trim().split('\n') : [];
      } catch {
        return [];
      }
    }
    ```
