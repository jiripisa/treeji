       ```typescript
         program
           .command('create <slug> <type>')
           .description('Create a worktree with branch {type}/{slug}')
           .action(async (slug: string, type: string) => { ... });
       }
       ```

    2. Slug sanitization — ALWAYS run user input through toSlug():
       ```typescript
       const cleanSlug = toSlug(slug);

         process.exit(1);
       }
       ```

    3. Path resolution — ALWAYS use git root (D-08, Pitfall 3):
       ```typescript
       import path from 'node:path';
       const gitRoot = await getGitRoot();

       ```

       ```typescript
       const branch = `${type}/${cleanSlug}`;
       ```

       ```typescript
       const spinner = p.spinner();

       await gitWorktreeAdd(worktreePath, branch);
       spinner.stop(`Worktree created at ${worktreePath}`);
       p.outro(`Branch: ${branch}`);
       ```

    6. Error handling: wrap in try/catch; on error call `p.cancel(err.message)` and `process.exit(1)`.
